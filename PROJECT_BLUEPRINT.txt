## 2. UTILITY LAYER

### Path: `/src/main/java/com/pcms/util/HibernateUtil.java`
```java
package com.pcms.util;
import org.hibernate.SessionFactory;
import org.hibernate.cfg.Configuration;

public class HibernateUtil {
    private static SessionFactory sessionFactory;
    static {
        try {
            Configuration config = new Configuration().configure("hibernate.cfg.xml");
            sessionFactory = config.buildSessionFactory();
        } catch (Exception e) { throw new ExceptionInInitializerError(e); }
    }
    public static SessionFactory getSessionFactory() { return sessionFactory; }
}
```

### Path: `/src/main/java/com/pcms/util/PasswordUtil.java`
```java
package com.pcms.util;
import java.security.MessageDigest;

public class PasswordUtil {
    public static String md5(String input) {
        try {
            MessageDigest md = MessageDigest.getInstance("MD5");
            byte[] hash = md.digest(input.getBytes());
            StringBuilder sb = new StringBuilder();
            for (byte b : hash) sb.append(String.format("%02x", b));
            return sb.toString();
        } catch (Exception e) { throw new RuntimeException(e); }
    }
}
```

package com.pcms.model;
import jakarta.persistence.*;
import java.util.Date;

@Entity @Table(name = "Requests")
public class Request {
    @Id @GeneratedValue(strategy = GenerationType.IDENTITY)
    private int id;
    private String firstName, lastName, email, telephone, unit, requestType;
    @Temporal(TemporalType.DATE) private Date requestDate;
    @Enumerated(EnumType.STRING) private RequestStatus status;
    @ManyToOne(fetch = FetchType.EAGER) @JoinColumn(name = "technicianId") private User technician;
    public enum RequestStatus { Pending, Technician_Assigned, Fixed, Not_fixed }
    public Request() { this.requestDate = new Date(); this.status = RequestStatus.Pending; }
    // Getters and Setters...
    public String getFullName() { return firstName + " " + lastName; }
}

## 4. DAO LAYER (Data Access)

### Path: `/src/main/java/com/pcms/dao/UserDAO.java`
```java
package com.pcms.dao;
import com.pcms.model.User;
import com.pcms.util.HibernateUtil;
import org.hibernate.Session;
import org.hibernate.query.Query;

public class UserDAO {
    public User login(String email, String password) {
        try (Session session = HibernateUtil.getSessionFactory().openSession()) {
            Query<User> q = session.createQuery("FROM User WHERE email = :email AND password = :pwd", User.class);
            q.setParameter("email", email); q.setParameter("pwd", password);
            return q.uniqueResult();
        }
    }
    public java.util.List<User> getAllTechnicians() {
        try (Session session = HibernateUtil.getSessionFactory().openSession()) {
            return session.createQuery("FROM User WHERE role = 'TECHNICIAN'", User.class).list();
        }
    }
}
```

## 5. SERVLET LAYER (MVC: C)

### Path: `/src/main/java/com/pcms/servlet/LoginServlet.java`
```java
package com.pcms.servlet;
import com.pcms.dao.UserDAO;
import com.pcms.model.User;
import com.pcms.util.PasswordUtil;
import jakarta.servlet.annotation.WebServlet;
import jakarta.servlet.http.*;
import java.io.IOException;

@WebServlet(urlPatterns = {"/", "/login"})
public class LoginServlet extends HttpServlet {
    private UserDAO userDAO = new UserDAO();

    protected void doGet(HttpServletRequest req, HttpServletResponse resp) throws IOException, jakarta.servlet.ServletException {
        req.getRequestDispatcher("/views/common/login.jsp").forward(req, resp);
    }

    protected void doPost(HttpServletRequest req, HttpServletResponse resp) throws IOException, jakarta.servlet.ServletException {
        String hashedPwd = PasswordUtil.md5(req.getParameter("password"));
        User user = userDAO.login(req.getParameter("email"), hashedPwd);
        if (user != null) {
            req.getSession().setAttribute("loggedUser", user);
            resp.sendRedirect(req.getContextPath() + (user.getRole() == User.Role.ADMIN ? "/admin/dashboard" : "/technician/dashboard"));
        } else {
            req.setAttribute("error", "Invalid credentials");
            req.getRequestDispatcher("/views/common/login.jsp").forward(req, resp);
        }
    }
}
```
